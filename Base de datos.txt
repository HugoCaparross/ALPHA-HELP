-- roles
CREATE TABLE IF NOT EXISTS public.roles (
  id          TEXT PRIMARY KEY,
  name        TEXT NOT NULL,
  description TEXT NOT NULL DEFAULT ''
);

-- interventions
CREATE TABLE IF NOT EXISTS public.interventions (
  id          UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name        TEXT UNIQUE NOT NULL,
  code        TEXT UNIQUE NOT NULL,
  description TEXT NOT NULL DEFAULT '',
  created_at  TIMESTAMPTZ DEFAULT NOW()
);

-- users (linked to auth.users)
CREATE TABLE IF NOT EXISTS public.users (
  id              UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  email           TEXT UNIQUE NOT NULL,
  full_name       TEXT NOT NULL DEFAULT '',
  role_id         TEXT NOT NULL DEFAULT 'participant' REFERENCES public.roles(id),
  intervention_id UUID REFERENCES public.interventions(id),
  email_verified  BOOLEAN NOT NULL DEFAULT FALSE,
  enrolled_at     TIMESTAMPTZ DEFAULT NOW(),
  created_at      TIMESTAMPTZ DEFAULT NOW(),
  updated_at      TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX IF NOT EXISTS idx_users_role         ON public.users(role_id);
CREATE INDEX IF NOT EXISTS idx_users_intervention ON public.users(intervention_id);

-- sessions
CREATE TABLE IF NOT EXISTS public.sessions (
  id              UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  intervention_id UUID NOT NULL REFERENCES public.interventions(id),
  title           TEXT NOT NULL,
  description     TEXT NOT NULL DEFAULT '',
  month_number    INT NOT NULL,
  youtube_live_url TEXT,
  live_at         TIMESTAMPTZ,
  recording_url   TEXT,
  is_published    BOOLEAN NOT NULL DEFAULT FALSE,
  created_at      TIMESTAMPTZ DEFAULT NOW(),
  updated_at      TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(intervention_id, month_number)
);
CREATE INDEX IF NOT EXISTS idx_sessions_intervention ON public.sessions(intervention_id, is_published);
CREATE INDEX IF NOT EXISTS idx_sessions_live         ON public.sessions(live_at) WHERE live_at IS NOT NULL;

-- recordings
CREATE TABLE IF NOT EXISTS public.recordings (
  id           UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  session_id   UUID NOT NULL REFERENCES public.sessions(id) ON DELETE CASCADE,
  url          TEXT NOT NULL,
  is_available BOOLEAN NOT NULL DEFAULT FALSE,
  created_at   TIMESTAMPTZ DEFAULT NOW()
);

-- session_access
CREATE TABLE IF NOT EXISTS public.session_access (
  id           UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id      UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  session_id   UUID NOT NULL REFERENCES public.sessions(id) ON DELETE CASCADE,
  is_unlocked  BOOLEAN NOT NULL DEFAULT FALSE,
  unlocked_at  TIMESTAMPTZ,
  UNIQUE(user_id, session_id)

CREATE INDEX IF NOT EXISTS idx_session_access_user ON public.session_access(user_id, is_unlocked);

-- helper: updated_at
CREATE OR REPLACE FUNCTION public.set_updated_at()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN NEW.updated_at = NOW(); RETURN NEW; END;
$$;

-- helper: is_admin (SECURITY DEFINER evita recursión en RLS)
CREATE OR REPLACE FUNCTION public.is_admin()
RETURNS BOOLEAN LANGUAGE sql SECURITY DEFINER STABLE SET search_path = public AS $$
  SELECT EXISTS (SELECT 1 FROM public.users WHERE id = auth.uid() AND role_id = 'admin');
$$;

-- create user profile on auth signup
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$
DECLARE v_region TEXT; v_int_id UUID;
BEGIN
  v_region := COALESCE(NEW.raw_user_meta_data->>'country', 'ES');

  SELECT id INTO v_int_id FROM public.interventions
  WHERE code = v_region LIMIT 1;

  -- fallback a España si no existe el código
  IF v_int_id IS NULL THEN
    SELECT id INTO v_int_id FROM public.interventions WHERE code = 'ES' LIMIT 1;
  END IF;

  INSERT INTO public.users(id, email, full_name, role_id, intervention_id, email_verified)
  VALUES (
    NEW.id, NEW.email,
    COALESCE(NEW.raw_user_meta_data->>'full_name', ''),
    'participant', v_int_id,
    (NEW.email_confirmed_at IS NOT NULL)
  )
  ON CONFLICT (id) DO NOTHING;

  INSERT INTO public.user_progress(user_id, completed_sessions_count)
  VALUES (NEW.id, 0)
  ON CONFLICT (user_id) DO NOTHING;

  -- desbloquear sesión mes 1 automáticamente
  IF v_int_id IS NOT NULL THEN
    INSERT INTO public.session_access(user_id, session_id, is_unlocked, unlocked_at)
    SELECT NEW.id, id, TRUE, NOW()
    FROM public.sessions
    WHERE intervention_id = v_int_id AND month_number = 1 AND is_published = TRUE
    ON CONFLICT (user_id, session_id) DO NOTHING;
  END IF;

  RETURN NEW;
END;
$$;

-- verify email
CREATE OR REPLACE FUNCTION public.verify_user_email(p_user_id UUID)
RETURNS VOID LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$
BEGIN
  UPDATE public.users SET email_verified = TRUE WHERE id = p_user_id;
END;
$$;

-- trigger: mark email verified when auth confirms it
CREATE OR REPLACE FUNCTION public.on_email_confirmed()
RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$
BEGIN
  IF NEW.email_confirmed_at IS NOT NULL AND OLD.email_confirmed_at IS NULL THEN
    PERFORM public.verify_user_email(NEW.id);
    INSERT INTO public.notifications(user_id, type, title, message)
    VALUES (NEW.id, 'system', '¡Bienvenido/a a Alpha-Help!',
      'Tu cuenta está activada. Accede a tu primera sesión cuando quieras.');
  END IF;
  RETURN NEW;
END;
$$;

-- assign intervention manually
CREATE OR REPLACE FUNCTION public.assign_intervention(p_user_id UUID, p_region TEXT)
RETURNS VOID LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$
DECLARE v_int_id UUID;
BEGIN
  SELECT id INTO v_int_id FROM public.interventions WHERE name = p_region;
  IF v_int_id IS NULL THEN
    RAISE EXCEPTION 'Intervención para la región % no encontrada.', p_region;
  END IF;

  UPDATE public.users SET intervention_id = v_int_id WHERE id = p_user_id;

  INSERT INTO public.session_access(user_id, session_id, is_unlocked, unlocked_at)
  SELECT p_user_id, id, TRUE, NOW()
  FROM public.sessions
  WHERE intervention_id = v_int_id AND month_number = 1 AND is_published = TRUE
  ON CONFLICT (user_id, session_id) DO NOTHING;
END;
$$;

-- unlock next session
CREATE OR REPLACE FUNCTION public.unlock_next_session(p_user_id UUID)
RETURNS BOOLEAN LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$
DECLARE
  v_int_id    UUID;
  v_count     INT;
  v_next_id   UUID;
BEGIN
  SELECT intervention_id INTO v_int_id FROM public.users WHERE id = p_user_id;
  SELECT completed_sessions_count INTO v_count FROM public.user_progress WHERE user_id = p_user_id;

  SELECT id INTO v_next_id FROM public.sessions
  WHERE intervention_id = v_int_id AND month_number = v_count + 1 AND is_published = TRUE;

  IF v_next_id IS NOT NULL THEN
    INSERT INTO public.session_access(user_id, session_id, is_unlocked, unlocked_at)
    VALUES (p_user_id, v_next_id, TRUE, NOW())
    ON CONFLICT (user_id, session_id)
    DO UPDATE SET is_unlocked = TRUE, unlocked_at = NOW();

    INSERT INTO public.notifications(user_id, type, title, message, link)
    VALUES (p_user_id, 'session', 'Nueva sesión disponible',
      'Tienes una nueva sesión desbloqueada en tu programa.', '/sesiones');

    RETURN TRUE;
  END IF;
  RETURN FALSE;
END;
$$;

-- create session access record
CREATE OR REPLACE FUNCTION public.create_session_access(p_user_id UUID, p_session_id UUID, p_unlocked BOOLEAN)
RETURNS VOID LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$
BEGIN
  INSERT INTO public.session_access(user_id, session_id, is_unlocked, unlocked_at)
  VALUES (p_user_id, p_session_id, p_unlocked, CASE WHEN p_unlocked THEN NOW() ELSE NULL END)
  ON CONFLICT (user_id, session_id) DO NOTHING;
END;
$$;

-- get available sessions for a user
CREATE OR REPLACE FUNCTION public.get_available_sessions(p_user_id UUID)
RETURNS TABLE (
  session_id   UUID,
  title        TEXT,
  month_number INT,
  is_unlocked  BOOLEAN
) LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$
BEGIN
  RETURN QUERY
  SELECT s.id, s.title, s.month_number,
    COALESCE(sa.is_unlocked, FALSE)
  FROM public.sessions s
  LEFT JOIN public.session_access sa
    ON s.id = sa.session_id AND sa.user_id = p_user_id
  WHERE s.intervention_id = (
    SELECT intervention_id FROM public.users WHERE id = p_user_id
  )
  AND s.is_published = TRUE
  ORDER BY s.month_number;
END;
$$;

-- update user progress after session completion
CREATE OR REPLACE FUNCTION public.update_user_progress(p_user_id UUID)
RETURNS VOID LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$
DECLARE v_count INT;
BEGIN
  SELECT COUNT(*) INTO v_count
  FROM public.user_sessions
  WHERE user_id = p_user_id AND is_completed = TRUE;

  UPDATE public.user_progress
  SET completed_sessions_count = v_count, last_activity_at = NOW()
  WHERE user_id = p_user_id;
END;
$$;

-- trigger wrapper for update_user_progress
CREATE OR REPLACE FUNCTION public.trigger_update_user_progress()
RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$
BEGIN
  IF NEW.is_completed = TRUE THEN
    PERFORM public.update_user_progress(NEW.user_id);
    PERFORM public.unlock_next_session(NEW.user_id);
  END IF;
  RETURN NEW;
END;
$$;

BEGIN

-- prevent duplicate completed questionnaire
CREATE OR REPLACE FUNCTION public.check_duplicate_questionnaire_attempt()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN
  IF EXISTS (
    SELECT 1 FROM public.questionnaire_attempts
    WHERE user_id = NEW.user_id
      AND questionnaire_type = NEW.questionnaire_type
      AND status = 'completed'
  ) THEN
    RAISE EXCEPTION 'Este cuestionario ya ha sido completado y no puede reenviarse.';
  END IF;
  RETURN NEW;
END;
$$;

-- validate password policy (frontend duplica esta validación)
CREATE OR REPLACE FUNCTION public.validate_password_policy(p_password TEXT)
RETURNS BOOLEAN LANGUAGE plpgsql IMMUTABLE AS $$
BEGIN
  RETURN length(p_password) >= 12
    AND p_password ~ '[A-Z]'
    AND p_password ~ '[a-z]'
    AND p_password ~ '[0-9]'
    AND p_password ~ '[!@#$%^&*(),.?":{}|<>]';
END;
$$;

-- validate questionnaire answers structure
CREATE OR REPLACE FUNCTION public.validate_questionnaire(p_questions JSONB, p_answers JSONB)
RETURNS BOOLEAN LANGUAGE plpgsql IMMUTABLE AS $$
BEGIN
  RETURN jsonb_typeof(p_answers) = 'object';
END;
$$;

-- audit log trigger function
CREATE OR REPLACE FUNCTION public.log_audit_action()
RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$
BEGIN
  INSERT INTO public.audit_logs(user_id, action, details)
  VALUES (
    auth.uid(),
    TG_OP || ' ON ' || TG_TABLE_NAME,
    'Row ID: ' || COALESCE(NEW.id::TEXT, OLD.id::TEXT)
  );
  RETURN COALESCE(NEW, OLD);
END;
$$;

-- admin stats
CREATE OR REPLACE FUNCTION public.admin_stats()
RETURNS TABLE (
  total_users        BIGINT,
  total_participants BIGINT,
  completed_pre      BIGINT,
  completed_post     BIGINT,
  spain_users        BIGINT,
  latam_users        BIGINT,
  unread_contacts    BIGINT
) LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$
BEGIN
  RETURN QUERY SELECT
    (SELECT COUNT(*)  FROM public.users),
    (SELECT COUNT(*)  FROM public.users WHERE role_id = 'participant'),
    (SELECT COUNT(*)  FROM public.questionnaire_attempts WHERE questionnaire_type = 'PRE'  AND status = 'completed'),
    (SELECT COUNT(*)  FROM public.questionnaire_attempts WHERE questionnaire_type = 'POST' AND status = 'completed'),
    (SELECT COUNT(*)  FROM public.users u JOIN public.interventions i ON u.intervention_id = i.id WHERE i.code = 'ES'),
    (SELECT COUNT(*)  FROM public.users u JOIN public.interventions i ON u.intervention_id = i.id WHERE i.code = 'LATAM'),
    (SELECT COUNT(*)  FROM public.contact_messages WHERE is_read = FALSE);
END;
$$;

-- ── users ─────────────────────────────────────────────────────
CREATE POLICY "users_own_select"
  ON public.users FOR SELECT
  USING (id = auth.uid() OR public.is_admin());

CREATE POLICY "users_own_update"
  ON public.users FOR UPDATE
  USING (id = auth.uid())
  WITH CHECK (id = auth.uid() AND role_id = 'participant');

CREATE POLICY "users_admin_all"
  ON public.users FOR ALL
  USING (public.is_admin());

-- ── sessions ──────────────────────────────────────────────────
CREATE POLICY "sessions_user_select"
  ON public.sessions FOR SELECT
  USING (
    is_published = TRUE
    AND intervention_id = (SELECT intervention_id FROM public.users WHERE id = auth.uid())
  );

CREATE POLICY "sessions_admin_all"
  ON public.sessions FOR ALL
  USING (public.is_admin());

-- ── session_access ────────────────────────────────────────────
CREATE POLICY "session_access_own_select"
  ON public.session_access FOR SELECT
  USING (user_id = auth.uid() OR public.is_admin());

CREATE POLICY "session_access_admin_all"
  ON public.session_access FOR ALL
  USING (public.is_admin());

-- ── user_sessions ─────────────────────────────────────────────
CREATE POLICY "user_sessions_own_select"
  ON public.user_sessions FOR SELECT
  USING (user_id = auth.uid() OR public.is_admin());

CREATE POLICY "user_sessions_own_insert"
  ON public.user_sessions FOR INSERT
  WITH CHECK (user_id = auth.uid());

CREATE POLICY "user_sessions_own_update"
  ON public.user_sessions FOR UPDATE
